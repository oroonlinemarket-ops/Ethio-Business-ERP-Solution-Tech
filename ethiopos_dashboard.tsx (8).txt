function Badge({label,color}){
  return <span style={{background:`${color}22`,color,fontSize:11,fontWeight:700,padding:"2px 8px",borderRadius:20}}>{label}</span>;
}
function StatRow({label,value,color,t}){
  return(
    <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",padding:"8px 0",borderBottom:`1px solid ${t.border}`}}>
      <span style={{fontSize:13,color:t.muted}}>{label}</span>
      <span style={{fontSize:13,fontWeight:700,color:color||t.text}}>{value}</span>
    </div>
  );
}
function Avatar({name,size=30,bg=C.light}){
  return(
    <div style={{width:size,height:size,borderRadius:"50%",background:`${bg}22`,display:"flex",alignItems:"center",justifyContent:"center",fontSize:size*0.38,fontWeight:700,color:bg,flexShrink:0}}>
      {name.split(" ").map(w=>w[0]).join("").slice(0,2)}
    </div>
  );
}
function TT({payload,active:isActive,t}){
  if(!isActive||!payload?.length)return null;
  return(
    <div style={{background:t?.card||"#1E293B",border:`1px solid ${t?.border||"#334155"}`,borderRadius:8,padding:"8px 12px",fontSize:12}}>
      {payload.map((p,i)=><div key={i} style={{color:p.color||"#94A3B8"}}>{p.name}: {p.value}</div>)}
    </div>
  );
}
function Tabs({tabs,active,set,t}){
  return(
    <div style={{display:"flex",gap:0,background:t.hover,borderRadius:10,padding:4,width:"fit-content",marginBottom:14}}>
      {tabs.map(([id,lbl])=>(
        <button key={id} onClick={()=>set(id)} style={{background:active===id?C.primary:"transparent",color:active===id?"#fff":t.muted,border:"none",borderRadius:8,padding:"7px 16px",cursor:"pointer",fontWeight:700,fontSize:13}}>
          {lbl}
        </button>
      ))}
    </div>
  );
}
function KPIGrid({items,t}){
  return(
    <div style={{display:"grid",gridTemplateColumns:`repeat(${items.length},1fr)`,gap:12,marginBottom:14}}>
      {items.map((s,i)=>(
        <div key={i} style={{background:t.card,border:`1px solid ${t.border}`,borderRadius:12,padding:14}}>
          <div style={{fontSize:11,color:t.muted}}>{s.l}</div>
          <div style={{fontSize:18,fontWeight:800,color:s.c||t.text,marginTop:4}}>{s.v}</div>
        </div>
      ))}
    </div>
  );
}
function Card({t,children,title,extraStyle}){
  return(
    <div style={{background:t.card,border:`1px solid ${t.border}`,borderRadius:14,padding:16,...(extraStyle||{})}}>
      {title&&<div style={{fontWeight:700,fontSize:14,marginBottom:12}}>{title}</div>}
      {children}
    </div>
  );
}

      case"warehouse":   return <WarehouseView t={t}/>;
      case"qa":          return <QADashboard t={t}/>;
      case"security":    return <SecurityView t={t}/>;
      case"plans":       return <SaaSPlans t={t}/>;
      case"help":        return <HelpCenter t={t}/>;
      case"readiness":   return <ReadinessReport t={t}/>;
      case"settings":    return <SettingsView t={t} darkMode={darkMode} setDarkMode={setDarkMode} lang={lang} setLang={setLang}/>;
      default:           return null;
    }
  }
  return(
    <div style={{display:"flex",height:"100vh",fontFamily:"'Segoe UI',system-ui,sans-serif",background:t.bg,color:t.text,overflow:"hidden"}}>
      <div style={{width:collapsed?52:210,background:t.sidebar,borderRight:"1px solid #1E293B",transition:"width 0.3s",display:"flex",flexDirection:"column",flexShrink:0}}>
        <div style={{padding:"12px 10px",display:"flex",alignItems:"center",gap:9,borderBottom:"1px solid #1E293B"}}>
          <div style={{width:30,height:30,borderRadius:8,background:`linear-gradient(135deg,${C.primary},${C.gold})`,display:"flex",alignItems:"center",justifyContent:"center",fontWeight:900,fontSize:13,color:"#fff",flexShrink:0}}>E</div>
          {!collapsed&&<div><div style={{fontWeight:800,fontSize:12,color:"#fff"}}>EthioPOS ERP</div><div style={{fontSize:9,color:"#475569"}}>21-Module Platform</div></div>}
        </div>
        {!collapsed&&(
          <div style={{padding:"7px 10px",borderBottom:"1px solid #1E293B",display:"flex",alignItems:"center",gap:7}}>
            <span style={{fontSize:13}}>{ROLES[role].icon}</span>
            <div>
              <div style={{fontSize:10,fontWeight:700,color:ROLES[role].color}}>{ROLES[role].label}</div>
              <button onClick={()=>setRole(null)} style={{background:"none",border:"none",color:"#475569",fontSize:9,cursor:"pointer",padding:0}}>Sign out</button>
            </div>
          </div>
        )}
        <nav style={{flex:1,overflowY:"auto",padding:"4px 0"}}>
          {nav.map(n=>(
            <button key={n.id} onClick={()=>setActive(n.id)} title={getLangLabel(n.id)} style={{width:"100%",display:"flex",alignItems:"center",gap:8,padding:"8px 12px",background:active===n.id?`${C.primary}22`:undefined,borderLeft:active===n.id?`3px solid ${C.light}`:"3px solid transparent",border:"none",cursor:"pointer",color:active===n.id?C.light:"#94A3B8",fontWeight:active===n.id?"700":"400",fontSize:12,textAlign:"left"}}>
              <span style={{fontSize:14,flexShrink:0}}>{n.icon}</span>
              {!collapsed&&<span>{getLangLabel(n.id)}</span>}
            </button>
          ))}
        </nav>
        <button onClick={()=>setCollapsed(p=>!p)} style={{background:"none",border:"none",color:"#64748B",padding:10,cursor:"pointer",fontSize:13}}>{collapsed?"→":"←"}</button>
      </div>
      <div style={{flex:1,display:"flex",flexDirection:"column",overflow:"hidden"}}>
        <div style={{background:t.card,borderBottom:`1px solid ${t.border}`,padding:"0 14px",height:50,display:"flex",alignItems:"center",justifyContent:"space-between",flexShrink:0}}>
          <div>
            <div style={{fontWeight:700,fontSize:14}}>{getLangLabel(active)}</div>
            <div style={{fontSize:10,color:t.muted}}>{branch} · {new Date().toLocaleDateString("en-ET",{weekday:"short",month:"short",day:"numeric"})}</div>
          </div>
          <div style={{display:"flex",gap:7,alignItems:"center"}}>
            <select value={branch} onChange={e=>setBranch(e.target.value)} style={{background:t.input,border:`1px solid ${t.border}`,color:t.text,borderRadius:7,padding:"4px 8px",fontSize:11,outline:"none",cursor:"pointer"}}>
              {["Bole Main","Merkato","Piassa","Bishoftu","Adama"].map(b=><option key={b}>{b}</option>)}
            </select>
            <select value={lang} onChange={e=>setLang(e.target.value)} style={{background:t.input,border:`1px solid ${t.border}`,color:t.text,borderRadius:7,padding:"4px 7px",fontSize:11,outline:"none",cursor:"pointer"}}>
              <option value="en">🇬🇧 EN</option>
              <option value="am">🇪🇹 AM</option>
              <option value="om">🟢 OM</option>
            </select>
            <button onClick={()=>setDarkMode(p=>!p)} style={{background:t.input,border:`1px solid ${t.border}`,color:t.text,borderRadius:7,padding:"4px 7px",cursor:"pointer",fontSize:13}}>{darkMode?"☀️":"🌙"}</button>
            <div ref={notifRef} style={{position:"relative"}}>
              <button onClick={()=>setNotifOpen(p=>!p)} style={{background:t.input,border:`1px solid ${t.border}`,color:t.text,borderRadius:7,width:32,height:32,cursor:"pointer",display:"flex",alignItems:"center",justifyContent:"center",fontSize:14,position:"relative"}}>
                🔔
                {unread>0&&<div style={{position:"absolute",top:-3,right:-3,width:15,height:15,borderRadius:"50%",background:C.danger,color:"#fff",fontSize:8,fontWeight:800,display:"flex",alignItems:"center",justifyContent:"center"}}>{unread}</div>}
              </button>
              {notifOpen&&(
                <div style={{position:"absolute",right:0,top:40,width:290,background:t.card,border:`1px solid ${t.border}`,borderRadius:12,boxShadow:"0 16px 40px #0005",zIndex:500}}>
                  <div style={{padding:"9px 14px",borderBottom:`1px solid ${t.border}`,display:"flex",justifyContent:"space-between"}}>
                    <div style={{fontWeight:700,fontSize:13}}>Notifications</div>
                    <button onClick={()=>setNotifs(p=>p.map(n=>({...n,read:true})))} style={{background:"none",border:"none",color:C.light,cursor:"pointer",fontSize:11}}>Mark all read</button>
                  </div>
                  {notifs.map(n=>(
                    <div key={n.id} onClick={()=>setNotifs(p=>p.map(x=>x.id===n.id?{...x,read:true}:x))} style={{padding:"9px 14px",borderBottom:`1px solid ${t.border}`,cursor:"pointer",background:n.read?"transparent":`${n.color}08`,display:"flex",gap:8}}>
                      <span style={{fontSize:15}}>{n.icon}</span>
                      <div style={{flex:1}}>
                        <div style={{fontSize:12,fontWeight:n.read?500:700}}>{n.title}</div>
                        <div style={{fontSize:11,color:t.muted,marginTop:1}}>{n.body}</div>
                      </div>
                      {!n.read&&<div style={{width:7,height:7,borderRadius:"50%",background:n.color,flexShrink:0,marginTop:4}}/>}
                    </div>
                  ))}
                </div>
              )}
            </div>
            <div style={{display:"flex",alignItems:"center",gap:6}}>
              <div style={{width:28,height:28,borderRadius:"50%",background:`linear-gradient(135deg,${C.primary},${C.purple})`,display:"flex",alignItems:"center",justifyContent:"center",fontWeight:700,color:"#fff",fontSize:12}}>{ROLES[role].icon}</div>
              <div style={{fontSize:10,fontWeight:600,color:ROLES[role].color}}>{ROLES[role].label}</div>
            </div>
          </div>
        </div>
        <div style={{flex:1,overflowY:"auto",padding:14}}>{renderPage()}</div>
      </div>
    </div>
  );
}
