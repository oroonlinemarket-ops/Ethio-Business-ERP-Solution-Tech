function MiniBar({ pct, color }) {
  return (
    <div style={{ background:"#334155", borderRadius:4, height:6, width:"100%", marginTop:4 }}>
      <div style={{ width:`${pct}%`, background:color, height:6, borderRadius:4, transition:"width 1s" }} />
    </div>
  );
}

function SalesChart({ theme }) {
  const max = Math.max(...weekSales);
  return (
    <div style={{ display:"flex", alignItems:"flex-end", gap:8, height:110, padding:"0 4px" }}>
      {weekSales.map((v, i) => {
        const h = Math.round((v/max)*100);
        const isToday = i===6;
        return (
          <div key={i} style={{ flex:1, display:"flex", flexDirection:"column", alignItems:"center", gap:4 }}>
            <div style={{ fontSize:9, color:theme.muted }}>{ETB(v/1000)}k</div>
            <div style={{
              width:"100%", height:`${h}%`,
              background: isToday ? `linear-gradient(180deg,${COLORS.primaryLight},${COLORS.primary})` : theme.border,
              borderRadius:"4px 4px 0 0",
              position:"relative", transition:"height 0.8s"
            }} />
            <div style={{ fontSize:10, color: isToday ? COLORS.primaryLight : theme.muted, fontWeight: isToday?"700":"400" }}>{weekLabels[i]}</div>
          </div>
        );
      })}
    </div>
  );
}

function POSView({ theme, lang }) {
  const items = [
    { name:"Cooking Oil 5L", price:250, cat:"Grocery" },
    { name:"Sugar 5kg", price:180, cat:"Grocery" },
    { name:"Rice 25kg", price:360, cat:"Grocery" },
    { name:"Teff 25kg", price:400, cat:"Grain" },
    { name:"Bread Loaf", price:25, cat:"Bakery" },
    { name:"Milk 1L", price:60, cat:"Dairy" },
  ];
  const [cart, setCart] = useState([]);
  const total = cart.reduce((a,c)=>a+c.price*c.qty,0);
  const add = item => setCart(p => {
    const ex = p.find(x=>x.name===item.name);
    return ex ? p.map(x=>x.name===item.name?{...x,qty:x.qty+1}:x) : [...p, {...item,qty:1}];
  });
  const remove = name => setCart(p=>p.map(x=>x.name===name&&x.qty>1?{...x,qty:x.qty-1}:x).filter(x=>x.qty>0));

  return (
    <div style={{ display:"flex", gap:16, height:"100%" }}>
      <div style={{ flex:1.2 }}>
        <div style={{ fontSize:12, color:theme.muted, marginBottom:10 }}>PRODUCTS — {lang==="am"?"ምርቶች":""}</div>
        <div style={{ display:"grid", gridTemplateColumns:"repeat(3,1fr)", gap:8 }}>
          {items.map(item => (
            <button key={item.name} onClick={()=>add(item)} style={{
              background:theme.card, border:`1px solid ${theme.border}`, borderRadius:10, padding:12,
              cursor:"pointer", textAlign:"left", color:theme.text
            }}>
              <div style={{ fontSize:12, fontWeight:600 }}>{item.name}</div>
              <div style={{ fontSize:11, color:theme.muted, marginTop:2 }}>{item.cat}</div>
              <div style={{ fontSize:14, fontWeight:700, color:COLORS.primaryLight, marginTop:6 }}>ETB {item.price}</div>

          ))}
        </div>
      </div>
      <div style={{ flex:1, background:theme.card, borderRadius:12, padding:16, border:`1px solid ${theme.border}`, display:"flex", flexDirection:"column" }}>

                <div style={{ display:"flex", alignItems:"center", gap:8 }}>
                  <div style={{ fontWeight:700, color:theme.text, fontSize:13 }}>ETB {c.price*c.qty}</div>
                  <button onClick={()=>remove(c.name)} style={{ background:COLORS.danger, color:"#fff", border:"none", borderRadius:4, padding:"2px 7px", cursor:"pointer", fontSize:12 }}>−</button>
                </div>
              </div>
            ))

          <div style={{ display:"flex", justifyContent:"space-between", fontWeight:700, fontSize:16, color:theme.text, marginBottom:12 }}>

function AIView({ theme, lang }) {
  const [msg, setMsg] = useState("");
  const [chat, setChat] = useState([
    { role:"ai", text:"Hello! I'm your AI Business Assistant. Ask me anything about your sales, inventory, or finances." }
  ]);
  const [loading, setLoading] = useState(false);

  const send = async () => {
    if (!msg.trim()) return;
    const userMsg = msg;
    setMsg("");
    setChat(p=>[...p,{role:"user",text:userMsg}]);
    setLoading(true);
    try {
      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method:"POST",
        headers:{"Content-Type":"application/json"},
        body: JSON.stringify({
          model:"claude-sonnet-4-20250514",
          max_tokens:1000,
          system:`You are an AI Business Assistant for EthioPOS, a POS and Business Management System for Ethiopian SMEs. 
          You help business owners with sales analytics, inventory management, financial insights, and business advice.
          The business uses Ethiopian Birr (ETB). Keep answers concise, practical, and relevant to Ethiopian business context.
          If asked in Amharic, respond in Amharic. Otherwise respond in English.`,
          messages:[
            ...chat.filter(c=>c.role!=="ai"||chat.indexOf(c)>0).map(c=>({role:c.role==="ai"?"assistant":"user",content:c.text})),
            {role:"user",content:userMsg}
          ]
        })
      });
      const data = await res.json();
      const reply = data.content?.find(b=>b.type==="text")?.text || "I couldn't process that. Please try again.";
      setChat(p=>[...p,{role:"ai",text:reply}]);
    } catch {
      setChat(p=>[...p,{role:"ai",text:"Connection error. Please check your internet and try again."}]);
    }
    setLoading(false);
  };

  return (
    <div style={{ display:"flex", flexDirection:"column", height:"100%", gap:16 }}>
      <div style={{ display:"grid", gridTemplateColumns:"repeat(2,1fr)", gap:10 }}>
        {aiInsights.map((ins,i)=>(
          <div key={i} style={{ background:theme.card, border:`1px solid ${theme.border}`, borderRadius:12, padding:14 }}>
            <div style={{ display:"flex", gap:8, alignItems:"center", marginBottom:6 }}>
              <span style={{ fontSize:18 }}>{ins.icon}</span>
              <span style={{ fontWeight:700, fontSize:13, color:ins.color }}>{ins.title}</span>
            </div>
            <div style={{ fontSize:12, color:theme.muted, lineHeight:1.5 }}>{ins.body}</div>
          </div>
        ))}
      </div>
      <div style={{ flex:1, background:theme.card, border:`1px solid ${theme.border}`, borderRadius:12, padding:16, display:"flex", flexDirection:"column" }}>
        <div style={{ fontWeight:700, fontSize:14, color:theme.text, marginBottom:12 }}>🤖 AI Business Assistant</div>
        <div style={{ flex:1, overflowY:"auto", display:"flex", flexDirection:"column", gap:8, marginBottom:12, minHeight:120 }}>
          {chat.map((c,i)=>(
            <div key={i} style={{ display:"flex", justifyContent:c.role==="user"?"flex-end":"flex-start" }}>
              <div style={{
                maxWidth:"80%", padding:"10px 14px", borderRadius:12, fontSize:13, lineHeight:1.5,
                background:c.role==="user"?COLORS.primary:theme.hover,
                color:c.role==="user"?"#fff":theme.text,
                borderBottomRightRadius:c.role==="user"?2:12,
                borderBottomLeftRadius:c.role==="ai"?2:12,
              }}>{c.text}</div>
            </div>
          ))}
          {loading && <div style={{ fontSize:12, color:theme.muted, alignSelf:"flex-start" }}>🤖 Thinking...</div>}
        </div>
        <div style={{ display:"flex", gap:8 }}>
          <input value={msg} onChange={e=>setMsg(e.target.value)}
            onKeyDown={e=>e.key==="Enter"&&send()}
            placeholder="Ask about sales, inventory, forecasts..."
            style={{ flex:1, background:theme.input, border:`1px solid ${theme.border}`, borderRadius:8, padding:"10px 14px", color:theme.text, fontSize:13, outline:"none" }}
          />
          <button onClick={send} style={{ background:COLORS.primary, color:"#fff", border:"none", borderRadius:8, padding:"10px 16px", fontWeight:700, cursor:"pointer", fontSize:13 }}>Send</button>
        </div>
      </div>
    </div>
  );
}

              <div style={{ fontWeight:700, fontSize:14, marginBottom:14 }}>📦 {L("Top Products Today","ዛሬ ምርጥ ምርቶች")}</div>
              {topProducts.map((p,i)=>(
                <div key={i} style={{ marginBottom:12 }}>
                  <div style={{ display:"flex", justifyContent:"space-between", fontSize:13 }}>
                    <span style={{ fontWeight:600 }}>{p.name}</span>
                    <span style={{ color:t.muted }}>ETB {p.rev.toLocaleString()}</span>
                  </div>
                  <div style={{ display:"flex", alignItems:"center", gap:8 }}>
                    <div style={{ flex:1 }}><MiniBar pct={p.pct} color={COLORS.primaryLight} /></div>
                    <span style={{ fontSize:11, color:t.muted, width:50, textAlign:"right" }}>{p.sold} sold</span>
                  </div>
                </div>
              ))}
            </div>
          </div>

          {/* Branches */}
          <div style={{ background:t.card, border:`1px solid ${t.border}`, borderRadius:14, padding:16 }}>
            <div style={{ fontWeight:700, fontSize:14, marginBottom:14 }}>🏪 {L("Branch Overview","ቅርንጫፍ አጠቃላይ")}</div>
            <div style={{ display:"grid", gridTemplateColumns:"repeat(4,1fr)", gap:10 }}>
              {branches.map((b,i)=>(
                <div key={i} style={{ background:t.hover, borderRadius:10, padding:14, border:`1px solid ${t.border}` }}>
                  <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:8 }}>
                    <span style={{ fontSize:12, fontWeight:700 }}>{b.name}</span>
                    <span style={{ width:8, height:8, borderRadius:"50%", background:b.status==="online"?COLORS.success:COLORS.danger, display:"inline-block" }} />
                  </div>
                  <div style={{ fontSize:15, fontWeight:800, color:t.text }}>{b.rev}</div>
                  <div style={{ fontSize:11, color:t.muted, marginTop:4 }}>{b.tx} transactions</div>
                </div>
              ))}
            </div>
          </div>
          </>
          )}
        </div>
      </div>
    </div>
  );
