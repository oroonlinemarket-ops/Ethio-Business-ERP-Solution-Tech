const useCountUp = (end, duration = 2000, start = false) => {
  const [count, setCount] = useState(0);
  useEffect(() => {
    if (!start) return;
    let s = 0; const step = end / (duration / 16);
    const t = setInterval(() => { s = Math.min(s + step, end); setCount(Math.round(s)); if (s >= end) clearInterval(t); }, 16);
    return () => clearInterval(t);
  }, [start, end, duration]);
  return count;
};

function StatCard({ value, label, start }) {
  const isNum = !isNaN(parseInt(value));
  const num = useCountUp(isNum ? parseInt(value) : 0, 1500, start);
  const prefix = value.includes("ETB") ? "ETB " : "";
  const suffix = value.includes("+") ? "+" : value.includes("%") ? "%" : "";
  return (
    <div style={{ textAlign:"center" }}>
      <div style={{ fontSize:36, fontWeight:900, color:"#fff" }}>{prefix}{isNum ? num : value}{suffix}</div>
      <div style={{ fontSize:13, color:"rgba(255,255,255,0.7)", marginTop:4 }}>{label}</div>
    </div>
  );
}
