# ============================================================
# EthioPOS — Full Production Docker Compose Stack
# Services: API · Frontend · PostgreSQL · Redis · Nginx · Backup
# ============================================================

# docker-compose.prod.yml
version: '3.9'

networks:
  ethiopos_net:
    driver: bridge

volumes:
  postgres_data:
  redis_data:
  nginx_certs:
  uploads:

services:

  # ── 1. PostgreSQL Database ─────────────────────────────────
  postgres:
    image: postgres:16-alpine
    container_name: ethiopos_db
    restart: unless-stopped
    environment:
      POSTGRES_DB:       ${DB_NAME:-ethiopos_db}
      POSTGRES_USER:     ${DB_USER:-ethiopos}
      POSTGRES_PASSWORD: ${DB_PASS}
      PGDATA:            /var/lib/postgresql/data/pgdata
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init.sql:/docker-entrypoint-initdb.d/init.sql:ro
    networks: [ethiopos_net]
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U ${DB_USER:-ethiopos}"]
      interval: 10s
      timeout: 5s
      retries: 5
    # Never expose 5432 to the internet — internal only
    expose: ["5432"]

  # ── 2. Redis (sessions, rate-limit, cache) ─────────────────
  redis:
    image: redis:7-alpine
    container_name: ethiopos_redis
    restart: unless-stopped
    command: redis-server --requirepass ${REDIS_PASS} --maxmemory 256mb --maxmemory-policy allkeys-lru
    volumes:
      - redis_data:/data
    networks: [ethiopos_net]
    expose: ["6379"]
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 3s
      retries: 3

  # ── 3. NestJS Backend API ──────────────────────────────────
  api:
    build:
      context: ./backend
      dockerfile: Dockerfile
      target: production
    container_name: ethiopos_api
    restart: unless-stopped
    depends_on:
      postgres: { condition: service_healthy }
      redis:    { condition: service_healthy }
    environment:
      NODE_ENV:          production
      PORT:              4000
      DB_HOST:           postgres
      DB_PORT:           5432
      DB_USER:           ${DB_USER}
      DB_PASS:           ${DB_PASS}
      DB_NAME:           ${DB_NAME}
      REDIS_URL:         redis://:${REDIS_PASS}@redis:6379
      JWT_SECRET:        ${JWT_SECRET}
      ANTHROPIC_API_KEY: ${ANTHROPIC_API_KEY}
      TELEBIRR_APP_ID:   ${TELEBIRR_APP_ID}
      TELEBIRR_APP_KEY:  ${TELEBIRR_APP_KEY}
      CBE_MERCHANT_ID:   ${CBE_MERCHANT_ID}
      CBE_API_KEY:       ${CBE_API_KEY}
      SMS_API_KEY:       ${SMS_API_KEY}
      ALLOWED_ORIGINS:   ${ALLOWED_ORIGINS}
      BACKUP_S3_BUCKET:  ${BACKUP_S3_BUCKET}
      AWS_REGION:        ${AWS_REGION:-eu-west-1}
    volumes:
      - uploads:/app/uploads
    networks: [ethiopos_net]
    expose: ["4000"]
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:4000/api/v1/health"]
      interval: 30s
      timeout: 10s
      retries: 3
    deploy:
      replicas: 2                    # horizontal scaling
      resources:
        limits:   { cpus: '1.0', memory: 512M }
        reservations: { cpus: '0.25', memory: 128M }

  # ── 4. Next.js Frontend ────────────────────────────────────
  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
      target: production
      args:
        NEXT_PUBLIC_API_URL: https://api.ethiopos.et/api/v1
    container_name: ethiopos_frontend
    restart: unless-stopped
    networks: [ethiopos_net]
    expose: ["3000"]
    environment:
      NEXT_PUBLIC_API_URL: https://api.ethiopos.et/api/v1
      NODE_ENV: production

  # ── 5. Nginx Reverse Proxy + SSL ──────────────────────────
  nginx:
    image: nginx:1.25-alpine
    container_name: ethiopos_nginx
    restart: unless-stopped
    depends_on: [api, frontend]
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - nginx_certs:/etc/letsencrypt
      - ./nginx/certbot:/var/www/certbot:ro
    networks: [ethiopos_net]

  # ── 6. Certbot (Let's Encrypt SSL renewal) ─────────────────
  certbot:
    image: certbot/certbot:latest
    container_name: ethiopos_certbot
    volumes:
      - nginx_certs:/etc/letsencrypt
      - ./nginx/certbot:/var/www/certbot
    command: >
      certonly --webroot -w /var/www/certbot
      --email admin@ethiopos.et
      --agree-tos --no-eff-email
      -d ethiopos.et -d app.ethiopos.et -d api.ethiopos.et
    profiles: ["ssl-init"]   # run manually: docker compose --profile ssl-init up certbot

  # ── 7. Automated Database Backup ──────────────────────────
  backup:
    image: postgres:16-alpine
    container_name: ethiopos_backup
    restart: unless-stopped
    depends_on:
      postgres: { condition: service_healthy }
    environment:
      PGPASSWORD:      ${DB_PASS}
      DB_HOST:         postgres
      DB_USER:         ${DB_USER}
      DB_NAME:         ${DB_NAME}
      S3_BUCKET:       ${BACKUP_S3_BUCKET}
      AWS_ACCESS_KEY:  ${AWS_ACCESS_KEY_ID}
      AWS_SECRET_KEY:  ${AWS_SECRET_ACCESS_KEY}
      AWS_REGION:      ${AWS_REGION:-eu-west-1}
    volumes:
      - ./scripts/backup.sh:/backup.sh:ro
    command: >
      sh -c "echo '0 3 * * * /backup.sh' | crontab - && crond -f"
    networks: [ethiopos_net]

---
